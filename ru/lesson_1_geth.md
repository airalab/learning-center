```js
var learning_center = web3.eth.contract(Core).at("0x4bdf1c4A800CE5408e0D95F52787C6924e07F5A3");
var lesson_addr = learning_center.get("Lesson_1");
var Lesson_1 = JSON.parse(learning_center.abiOf(lesson_addr));
```
